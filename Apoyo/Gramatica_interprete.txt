
INICIO ->			PROGRAM id PTYCOMA <INFERIORES> <ADENTRO>
					|PROGRAM id PTYCOMA <ADENTRO>
<ADENTRO> ->		BEGIN <CODIGO> END .
					|BEGIN END .

<INFERIORES> ->		|<INFERIORES><INFERIOR>
					|<INFERIOR>

<INFERIOR> ->		<DECLARACIONES>
					|<CONST>
					|<TYPE>
					|<FUNCION>
					|<PROCEDIMIENTO>


<DECLARACIONES> ->	VAR <GR_VARS>

<GR_VARS> ->		<GR_VARS> <GR_VAR>
					|<GR_VAR>

<GR_VAR> ->			<LISTA_IDS> //listado de variables
					|<VAR_DECL> //variables con declaracion

<LISTA_IDS> ->		<LISTA_IDS> id PTYCOMA
					|id DOSPUNTOS <TIPO> PTYCOMA

<VAR_DECL> ->		id DOSPUNTOS <TIPO> IGUAL <EXPRESION> PTYCOMA

<TIPO> ->			CHAR
					|INTEGER
					|REAL
					|BOOLEAN
					|STRING

<CONST> ->			CONST id IGUAL <EXPRESION> PTYCOMA

<FUNCION> ->		FUNCTION id PARIZQ <PARAMETROS> PARDER DOSPUNTOS <TIPO> PTYCOMA <LOCAL_DECL> BEGIN <ADENTRO> END PTYCOMA
					|FUNCTION id PARIZQ <PARAMETROS> PARDER DOSPUNTOS <TIPO> PTYCOMA <LOCAL_DECL> BEGIN END PTYCOMA
					|FUNCTION id PARIZQ <PARAMETROS> PARDER DOSPUNTOS <TIPO> PTYCOMA BEGIN <ADENTRO> END PTYCOMA
					|FUNCTION id PARIZQ <PARAMETROS> PARDER DOSPUNTOS <TIPO> PTYCOMA BEGIN END PTYCOMA
					|FUNCTION id PARIZQ PARDER DOSPUNTOS <TIPO> PTYCOMA <LOCAL_DECL> BEGIN <ADENTRO> PTYCOMA END PTYCOMA
					|FUNCTION id PARIZQ PARDER DOSPUNTOS <TIPO> PTYCOMA <LOCAL_DECL> BEGIN END PTYCOMA
					|FUNCTION id PARIZQ PARDER DOSPUNTOS <TIPO> PTYCOMA BEGIN <ADENTRO> END PTYCOMA
					|FUNCTION id PARIZQ PARDER DOSPUNTOS <TIPO> PTYCOMA BEGIN END PTYCOMA

<PARAMETROS> ->		<PARAMETROS> <PARAMETRO> PTYCOMA
					|<PARAMETRO>

<PARAMETRO> ->		<X_VALOR>
					|var <X_REFERENCIA>

<X_VALOR> ->		<X_VALOR> id COMA
					|id DOSPUNTOS <TIPO>

<X_REFERENCIA> ->	<X_REFERENCIA> id COMA
					|id DOSPUNTOS <TIPO>
